22 {
23  using namespace std;

190 inline bool startsWith(const char* str, const char* substr)
191 {
192  int current = 0;
193  while (str[current] && substr[current])
194  {
195  if (str[current] != substr[current]) { return false; }
196  ++current;
197  }
198  if (!str[current] && substr[current]) { return false; }
199  return true;

223 /*!***************************************************************************************************************
224 \brief Tests if a string ends with another string.
225 \param str The string whose end will be checked. Not null terminated - length is passed explicitly.
226 \param lenStr The length of str
227 \param substr The sequence of characters to check if str ends with. Not null terminated - length is passed
228  explicitly.
229 \param lenSubstr The length of Substr
230 \return true if the string substr is indeed the last characters of str, false otherwise.
231 *****************************************************************************************************************/
232 inline bool endsWith(const char* str, int32 lenStr, const char* substr, int32 lenSubstr)
233 {
234  if (lenSubstr > lenStr || !lenStr--) { return false; }
235  if (!lenSubstr--) { return true; }
236  while (lenStr >= 0 && lenSubstr >= 0)
237  {
238  if (str[lenStr--] != substr[lenSubstr--]) { return false; }
239  }
240  if (!lenStr && lenSubstr) { return false; }
241  return true;
242 }

277 /*!****************************************************************************************************************
278 \brief Separate a filename to name and extension
279 \param[in] fileAndExtension A filename
280 \param[out] filename The file part of the name (part before the last '.')
281 \param[out] extension The extension part of the name (part after the last '.')
282 \description The period is returned in neither filename nor the extension
283 *******************************************************************************************************************/
284 inline void getFileNameAndExtension(const std::string& fileAndExtension, std::string& filename, std::string& extension)
285 {
286  auto it = std::find(fileAndExtension.rbegin(), fileAndExtension.rend(), '.');
287  if (it == fileAndExtension.rend())
288  {
289  filename = fileAndExtension;
290  extension = "";
291  return;
292  }
293  size_t position = fileAndExtension.rend() - it; //rend is the position one-after the start of the string.
294  filename.assign(fileAndExtension.begin(), fileAndExtension.begin() + position - 1);
295  extension.assign(fileAndExtension.begin() + position, fileAndExtension.end());
296 }
