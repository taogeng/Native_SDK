18 /*!********************************************************************************************************************
19 \brief This class provides functionality to handle the volume enclosed by 6 planes, normally the viewing frustum.
20 The planes are represented in Hessian Normal Form (normal, distance) as vec4( (xyz):[normal], w:[distance from 0,0,0] )
21 In order to optimize the frustum, the following are assumed:
22 A "frustum part" of a plane means "the quadrilateral of the plane that is enclosed by the 4 other planes not opposite to it"
23 1) Frustum parts opposite each other do not intersect
24 2) Any point of a positive(negative) part of the frustum has a larger(smaller) corresponding coordinate than its opposite part
25 3) The frustum is "opening" accross the z axis accross al direction, meaning that the positive Z part of the frustum
26  completely encloses the projection of the negative z part on it.
27  These optimizations allow us to
28 **********************************************************************************************************************/

39 /*!********************************************************************************************************************
40 \brief This class provides specialized functionality for when a frustum is a "normal"viewing frustum, that is, the following
41 conditions hold (The conditions ARE NOT checked)
42 Note: A "frustum side" of a plane means "the quadrilateral of the plane that is enclosed by the 4 other planes not opposite to it"
43 "Opposite" means the "other" side of the frustum (minusX is opposit plusX).
44 1) Opposite Frustum sides do not intersect (their planes may do so outside the frustum)
45 2) The frustum is "opening", or at least not "closing" accross the z axis accross al direction, meaning that the any point of the
46 projection of the negative Z side of the frustum on the positive Z plane, is inside or on the positive Z side of the frustum.
47 3) Any point of a positive(negative) part of the frustum has a larger(smaller) corresponding coordinate than its opposite part
48 These optimizations allow us to greatly reduce the calculations for a viewing frustum.
49 4) All plane normals point INTO of the frustum
50 **********************************************************************************************************************/
51 struct ViewingFrustum : public Frustum

230  /*!****************************************************************************************************************
231  \brief Get the local bounding box transformed by a provided 3x3 matrix and translation vector.
232  \param m Rotation and scale matrix.
233  \param t Translation vector
234  \param[out] outAABB The transformed AABB will be stored here. Previous contents ignored.
235  *******************************************************************************************************************/
236  void transform(const glm::mat3& m, const glm::vec3& t, AxisAlignedBox& outAABB)
237  {
238  for (int i = 0; i < 3; i++)
239  {
240  outAABB.m_center[i] = t[i];
241  outAABB.m_halfExtent[i] = 0.0f;
242 
243  outAABB.m_center[i] += m[i][0] * m_center[0];
244  outAABB.m_halfExtent[i] += fabs(m[i][0]) * m_halfExtent[0];
245 
246  outAABB.m_center[i] += m[i][1] * m_center[1];
247  outAABB.m_halfExtent[i] += fabs(m[i][1]) * m_halfExtent[1];
248 
249  outAABB.m_center[i] += m[i][2] * m_center[2];
250  outAABB.m_halfExtent[i] += fabs(m[i][2]) * m_halfExtent[2];
251  }
252  }
