15 /*!*********************************************************************************************************************
16 \brief A combination of array (std::vector) with associative container (std::map). Supports association of names
17 with values, and retrieval by index.
18 \description An std::vector style array class with the additional feature of associating "names" (IndexType_, std::string
19 by default) with the values stored. Keys are of type IndexType_, correspond to vector position 1:1, so that each vector position
20 ("index") is associated with a "key", and only one key.
21 Use: Add pairs of values with insert(key, value).
22 Retrieve indices by key, using getIndex(key) -- O(logn)
23 Retrieve values by index, using indexing operator [] -- O(1)
24 The remove() function destroys the items on which it was called, but a default-constructed object will still exist. Performing
25 insert() after removing an item will use the place of a previously deleted item, if it exists.
26 CAUTION: If remove() has been called, the vector no longer guarantees contiguousness until compact() is called.
27 CAUTION: To manually reclaim all memory and guarantee contiguous allocation, call compact(). Calling compact invalidates all
28 indexes, which must then be retrieved anew by "getInxdex".
29 Calling getIndex on an unknown key returns (size_t)(-1)
30 Accessing an unknown item by index is undefined.
31 Accessing an index not retrieved by getIndex since the last compact() operation is undefined.
32 ***********************************************************************************************************************/

395  /*!*********************************************************************************************************************
396  \brief Compacts the backing array by removing existing items from the end of the vector and putting them in the place of
397  deleted items, and then updating their index, until no more positions marked as deleted are left. Will ensure the
398  contiguousness of the backing vector, but will invalidate previously gotten item indexes.
399  ***********************************************************************************************************************/
400  void compact()
401  {
402  //We can do that because the last remove() tears down all datastructures used.
403  if (!myindex.size())
404  {
405  return;
406  }
407  //First, make sure there is something to compact...
408  deleteditemlisttype_::iterator unused_spot = myDeletedItems.begin();
409  while (myDeletedItems.size() && unused_spot != myDeletedItems.end())
410  {
411  //Last item in the storage vector.
412  size_t last = mystorage.size();
413 
414  //1) Trim the end of the vector...
415  while (last-- && mystorage[last].isUnused)
416  {
417  mystorage.pop_back();
418  //Rinse, repeat. If size is zero, there is nothing to do.
419  }
420 
421  //Either the storage is empty
422  if (mystorage.empty())
423  {
424  myDeletedItems.clear();
425  //the rest should already be cleared...
426  //the loop will exit naturally
427  }
428  else //Or we can have find its last item!
429  {
430  //2)Trim any items from the end of the unused spots list that may have been trimmed off by the vector...
431  while (unused_spot != myDeletedItems.end() && *unused_spot >= last)
432  {
433  myDeletedItems.erase(unused_spot++);
434  }
435  //Any spots left?
436  if (unused_spot != myDeletedItems.end())
437  {
438  //Do the actual data movement. After all we've been through, we know that
439  //i. The last item of the vector is a valid item (guaranteed by 1)
440  //ii. The unused spot is not out of bounds of the vector
441  //iii.
442  //Copy by hand(as we have not defined a move assignment operator for compatibility reasons.
443  //Also, since the string does not throw, this makes easier to reason about exceptions.
444 #ifdef PVR_SUPPORT_MOVE_SEMANTICS
445  mystorage[*unused_spot].value = std::move(mystorage[last].value);
446  mystorage[*unused_spot].key = std::move(mystorage[last].key);
447 #else
448  mystorage[*unused_spot].value = mystorage[last].value;
449  mystorage[last].value.ValueType_::~ValueType_();
450  new(&mystorage[last].value) ValueType_();
451  mystorage[*unused_spot].key = mystorage[last].key;
452  mystorage[last].key.clear();
453 #endif
454  mystorage[*unused_spot].isUnused = false;
455  myindex[mystorage[*unused_spot].key] = *unused_spot;
456  mystorage.pop_back();
457  myDeletedItems.erase(unused_spot++);
458  }//else : No action needed - unused spots has been trimmed off completely, so no movement is possible, or necessary...
459  }
460  }
461  }

509  /*!*********************************************************************************************************************
510  \brief Move a specific item (identified by a key) to a specific index in the list. If an item is already in this spot in the
511  list, their positions are swapped.
512  \return False if the specified key was not found in the index.
513  ***********************************************************************************************************************/
514  bool relocate(const IndexType_& key, size_t index)
515  {
516  typename maptype_::iterator found = myindex.find(key);
517  if (found == myindex.end()) { return false; }
518  size_t old_index = myindex[key];
519  if (index == old_index) { return true; } //No-op
520  if (index + 1 > mystorage.size()) // Storage not big enough.
521  {
522  //Grow, and mark unused all required items. Need to add all spots (But the last) to unusedspots.
523  size_t oldsize = mystorage.size();
524  mystorage.resize(index + 1);
525  for (size_t i = oldsize; i < index; ++i)
526  {
527  myDeletedItems.push_front(i);
528  mystorage[i].isUnused = 1;
529  }
530  mystorage.back() = mystorage[old_index];
531  removefromvector(old_index);
532  }
533  else if (mystorage[index].isUnused) // Lucky! Storage is big enough, and the item is not used!
534  {
535  deleteditemlisttype_::iterator place = std::find(myDeletedItems.begin(), myDeletedItems.end(), index);
536  assert(place != myDeletedItems.end()); // Shouldn't happen! Ever!
537  myDeletedItems.erase(place);
538  mystorage[index] = mystorage[old_index];
539  removefromvector(old_index);
540  }
541  else // Whoops! Space is already occupied. Swap with the old item!
542  {
543  myindex[mystorage[index].key] = old_index;
544  std::swap(mystorage[index], mystorage[old_index]);
545  }
546  myindex[key] = index;
547  return true;
548  }

551  size_t insertinvector(const IndexType_& key, const ValueType_ & val)
552  {
553  size_t retval;
554  if (myDeletedItems.empty())
555  {
556  retval = mystorage.size();
557  mystorage.push_back(StorageItem_());
558  mystorage.back().isUnused = false;
559  mystorage.back().key = key;
560  mystorage.back().value = val;
561  }
562  else
563  {
564  retval = myDeletedItems.back();
565  myDeletedItems.pop_back();
566  mystorage[retval].value = val;
567  mystorage[retval].key = key;
568  mystorage[retval].isUnused = false;
569  }
570  return retval;
571  }
572  void removefromvector(size_t index)
573  {
574  if (index == (mystorage.size() - 1))
575  {
576  //Removing the last item from the vector -- just pop it.
577  mystorage.pop_back();
578  }
579  else
580  {
581  //NOT the last item, so we just destruct it (to free any potential expensive resources),
582  //and then default-construct it (to have the spot destructible). We keep the reference to
583  //the key as we will need it.
584  myDeletedItems.push_front(index);
585  mystorage[index].isUnused = true;
586  mystorage[index].value.ValueType_::~ValueType_();
587  new(&mystorage[index].value) ValueType_;
588  }
