111  /*!********************************************************************************************************
112  \brief Convenience functions that reads all data in the stream into a contiguous block of memory of
113  a specified element type.
114  Requires random-access stream.
115  \tparam Type_ The type of item that will be read into.
116  \return A std::vector<Type_> containing all data from the current point to the end of the stream.
117  **********************************************************************************************************/
118  template<typename Type_> std::vector<Type_> readToEnd() const
119  {
120  std::vector<Type_> ret;
121  size_t mySize = getSize();
122  size_t myElements = mySize / sizeof(Type_);
123  ret.resize(mySize);
124  size_t actuallyRead;
125  read(sizeof(Type_), myElements, ret.data(), actuallyRead);
126  return ret;
127  }
