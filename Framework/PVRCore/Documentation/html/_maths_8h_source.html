121 \brief Performs quadratic interpolation between two points, beginning with a faster rate and slowing down.
122 \param start The starting point.
123 \param end The end point
124 \param factor Interpolation factor. At 0, returns start. At 1, returns end. Closer to 0, the rate of change is faster, closer to 1 slower.

126 inline float quadraticEaseOut(pvr::float32 start, pvr::float32 end, pvr::float32 factor)

178 /*!********************************************************************************************
179 \brief Calculated a tilted projection matrix
180 \param fovy The field of vision in the y axis
181 \param aspect The aspect of the viewport
182 \param near1 The near clipping plane distance
183 \param far1 The far clipping plane distance
184 \param rotate Angle of tilt (rotation around the z axis), in radians
185 \return A projection matrix for the specified parameters, tilted by rotate
186 ***********************************************************************************************/
187 inline glm::mat4 perspective(pvr::Api::Enum api, float32 fovy, float32 aspect, float32 near1,
188  float32 far1, pvr::float32 rotate = .0f)
189 {
190  glm::mat4 mat = glm::perspective(fovy, aspect, near1, far1);
191  if (api == Api::Vulkan)
192  {
193  mat[1][1] *= -1.f;// negate the y axis's y component, because vulkan coordinate system is +y down
194  }
195  return (rotate == 0.f ? mat : glm::rotate(rotate, glm::vec3(0.0f, 0.0f, 1.0f)) * mat);
196 }
