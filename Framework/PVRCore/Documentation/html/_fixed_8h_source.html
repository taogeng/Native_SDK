371 template <typename IntegerType, bool IsSigned>
372 void FixedPoint<IntegerType, IsSigned>::packDouble(double value)
373 {
374  // Check if the value needs to be clamped.
375  if (value >= getMaximumAbsoluteValue().unpackDouble() )
376  {
377  m_value = getMaximumAbsoluteValue().m_value;
378  }
379  else if (value <= getLowestValue().unpackDouble() )
380  {
381  m_value = getLowestValue().m_value;
382  }
383  else
384  {
385  if (IsSigned)
386  {
387  // Access the bit fields of the storage type directly.
388  SignedBitFields& bitFields = reinterpret_cast<SignedBitFields&>(m_value);
389 
390  double absoluteValue = fabs(value);
391 
392  bitFields.signBit = (value < 0) ? 1 : 0;
393  bitFields.integralPart = (static_cast<IntegerType>(absoluteValue) - bitFields.signBit);
394  bitFields.fractionalPart = static_cast<IntegerType>( (absoluteValue - floor(absoluteValue) ) * static_cast<double>(c_multiplier) );
395  }
396  else
397  {
398  // Access the bit fields of the storage type directly.
399  BitFields& bitFields = reinterpret_cast<BitFields&>(m_value);
400 
401  bitFields.fractionalPart = static_cast<IntegerType>( (value - floor(value) ) * static_cast<double>(c_multiplier) );
402  }
403  }
404 }

406 template <typename IntegerType, bool IsSigned>
407 double FixedPoint<IntegerType, IsSigned>::unpackDouble() const
408 {
409  // If it's signed, need special behaviour to handle the sign bit.
410  if (IsSigned)
411  {
412  SignedBitFields bitFields = reinterpret_cast<const SignedBitFields&>(m_value);
413 
414  double result = static_cast<double>(bitFields.fractionalPart) / static_cast<double>(c_multiplier);
415 
416  result += static_cast<double>(bitFields.integralPart);
417 
418  result += bitFields.signBit;
419 
420  result *= bitFields.signBit ? -1.0 : 1.0;
421 
422  return result;
423  }
424  else
425  {
426  BitFields bitFields = reinterpret_cast<const BitFields&>(m_value);
427 
428  double result = static_cast<double>(bitFields.fractionalPart) / static_cast<double>(c_multiplier);
429 
430  result += static_cast<double>(bitFields.integralPart);
431 
432  return result;
433  }
