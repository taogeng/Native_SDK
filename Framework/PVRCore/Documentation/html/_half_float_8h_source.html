127  void packFloat(const float& value)
128  {
129  HalfFloatData halfData = { m_value };
130  //Get the raw bits of the floating point value.
131  const FloatData floatFields = { value };
132 
133  //Work out the new exponent.
134  int newExponent = floatFields.bits.exponent - c_floatExponentBias + c_halfExponentBias;
135 
136  //Set a sort of "default state" for the half float value
137  halfData.value = 0;
138  halfData.bits.sign = floatFields.bits.sign;
139 
140  //Check for NaNs, Infinities, Overflows and Underflows.
141 
142  //Anything above or equal to halfExponentNaN is going to be either too large to represent or a NaN.
143  if (newExponent >= c_halfExponentNaN)
144  {
145  //Any exponent greater than the representable range will return Infinity.
146  halfData.bits.exponent = c_halfExponentNaN;
147 
148  //Ensure that any NaNs are still a NaN, and not accidentally turned into an infinity.
149  if (floatFields.bits.exponent == c_floatExponentNaN && floatFields.bits.mantissa != 0)
150  {
151  halfData.bits.mantissa = 0x200;
152  }
153  }
154  //Underflows may be representable with a loss of accuracy
155  else if (newExponent <= 0)
156  {
157  //Attempt to deal with it if we can.
158  if (newExponent >= -10)
159  {
160  //Calculate the new mantissa, masking off the top bit.
161  unsigned int newMantissa = floatFields.bits.mantissa | 0x800000;
162  halfData.bits.mantissa = newMantissa >> (14 - newExponent);
163 
164  //Check if it needs to be rounded up
165  if ((newMantissa >> (13 - newExponent)) & 1)
166  {
167  //It might overflow to the exponent bit, but this is fine due to the way floats work.
168  halfData.value++;
169  }
170  }
171  }
172  else
173  {
174  //Simple case, just set the exponent and mantissa.
175  halfData.bits.exponent = newExponent;
176  halfData.bits.mantissa = floatFields.bits.mantissa >> 13;
177 
178  //Check if it needs to be rounded up
179  if (floatFields.bits.mantissa & 0x1000)
180  {
181  //It might overflow to the exponent bit, but this is fine due to the way floats work.
182  halfData.value++;
183  }
184  }

188  float unpackFloat() const
189  {
190  //Get the raw bits of the half float, and turn them into a float.
191  FloatData floatFields;
192  const HalfFloatData halfData = { m_value };
193  //Set a default value for the float
194  floatFields.bits.sign = halfData.bits.sign;
195  floatFields.bits.exponent = 0;
196  floatFields.bits.mantissa = 0;
197 
198  // Check for anything that isn't signed zero
199  if (halfData.bits.exponent != 0 || halfData.bits.mantissa != 0)
200  {
201  //Some de-normalised half floats can be normalised - due to the increased mantissa of a full float.
202  //This is a kind of slow operation, but means that the floating point values can be easier to work on in hardware and thus faster.
203  if (halfData.bits.exponent == 0)
204  {
205  // Adjust the exponent and mantissa to try to normalise the new value.
206  unsigned int exponentAdjustment = 0;
207  unsigned int newMantissa = halfData.bits.mantissa << 1;
208 
209  //Loop through until the mantissa hits its maximum
210  while ((newMantissa & 0x400) == 0)
211  {
212  exponentAdjustment++;
213  newMantissa <<= 1;
214  }
215 
216  //Set the new mantissa and exponent values
217  floatFields.bits.mantissa = (newMantissa & 0x3ff) << 13;
218  floatFields.bits.exponent = (c_floatExponentBias - c_halfExponentBias) - exponentAdjustment;
219  }
220  //Handle infinities and NaNs
221  else if (halfData.bits.exponent == c_halfExponentNaN)
222  {
223  //Expand the mantissa up to the original size, and set the NaN value for the exponent.
224  floatFields.bits.exponent = c_floatExponentNaN;
225  floatFields.bits.mantissa = halfData.bits.mantissa << 13;
226  }
227  //Otherwise it's a simple normalised number
228  else
229  {
230  floatFields.bits.exponent = halfData.bits.exponent + (c_floatExponentBias - c_halfExponentBias);
231  floatFields.bits.mantissa = halfData.bits.mantissa << 13;
232  }
233  }
234 
235  return floatFields.value;
236  }
