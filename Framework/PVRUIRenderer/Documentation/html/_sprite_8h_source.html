741 /*!****************************************************************************************************************
742 \brief Abstract container for sprites. See MatrixGroup or PixelGroup. A group contains references to a number
743  of sprites, allowing hierarchical transformations to be applied to them
744 \details A very complex transformation using the "id" member is used to optimize the group transformations, as
745  each child needs to hold the transformations
746 *******************************************************************************************************************/
747 class Group_ : public Sprite_
748 {
749 protected:
751  struct SpriteEntryEquals
752  {
753  Sprite sprite;
754  SpriteEntryEquals(const Sprite& sprite) : sprite(sprite) {}
755  bool operator()(const Sprite& rhs) { return sprite == rhs; }
756  };
757  typedef std::vector<Sprite> ChildContainer;
758  mutable ChildContainer m_children;
759  pvr::uint64 m_id;
761 public:
762  /*!****************************************************************************************************************
763  \brief Constructor. Internal use. The parameter groupid is an implementation detail used to implement and optimize
764  group behaviour, and cannot be trivially determined.
765  *******************************************************************************************************************/
766  Group_(UIRenderer& uiRenderer, pvr::uint64 groupid) : Sprite_(uiRenderer), m_id(groupid) { }
767 
768  /*!****************************************************************************************************************
769  \brief Add a Sprite (Text, Image etc.) to this Group. All sprites in the group will be transformed together when
770  calling Render on the group.
771  \param sprite The Sprite to add.
772  \return Pointer to this object, in order to easily chan add commands.
773  *******************************************************************************************************************/
774  Group_* add(const Sprite& sprite)
775  {
776  m_children.push_back(sprite);
777  m_boundingRect.add(sprite->getDimensions().x, sprite->getDimensions().y, 0.0f);
778  return this;
779  }
780 
781  /*!****************************************************************************************************************
782  \brief Remove a Sprite from this Group. Linear search (Complexity O(n) )
783  \param sprite The Sprite to remove.
784  *******************************************************************************************************************/
785  void remove(const Sprite& sprite)
786  {
787  ChildContainer::iterator it = std::find_if(m_children.begin(), m_children.end(), SpriteEntryEquals(sprite));
788  if (it != m_children.end())
789  {
790  m_boundingRect.remove((*it)->getBoundingBox());
791  m_children.erase(it);
792  }
793  }
794 
795  /*!****************************************************************************************************************
796  \internal
797  \brief Internal function. Do not call directly. Triggered when all item transformations have been applied and the
798  final matrices can be calculated.
799  *******************************************************************************************************************/
800  void calculateMvp(pvr::uint64 parentIds, glm::mat4 const& srt, const glm::mat4& viewProj,
801  pvr::Rectanglei const& viewport)const
802  {
803  glm::mat4 tmpMatrix = srt * m_cachedMatrix;
804  //My cached matrix should always be up-to-date unless overridden. No effect.
805  for (ChildContainer::iterator it = m_children.begin(); it != m_children.end(); ++it)
806  {
807  (*it)->calculateMvp(packId(parentIds, m_id), tmpMatrix, viewProj, viewport);
808  }
809  }
810 
811 
812  /*!****************************************************************************************************************
813  \internal
814  \brief Internal function that UIRenderer calls to render. Do not call directly.
815  *******************************************************************************************************************/
816  virtual void onRender(api::SecondaryCommandBuffer& commandBuffer, pvr::uint64 parentId) const
817  {
818  for (ChildContainer::iterator it = m_children.begin(); it != m_children.end(); ++it)
819  {
820  (*it)->onRender(commandBuffer, packId(parentId, m_id));
821  }
822  }
823 protected:
824  pvr::uint64 packId(pvr::uint64 parentIds, pvr::uint64 id)const
825  {
826  pvr::uint64 packed = parentIds << NUM_BITS_GROUP_ID;
827  return packed |= id;
828  }
829 };
