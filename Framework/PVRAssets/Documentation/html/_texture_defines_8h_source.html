100 /*!****************************************************************************************
101 \brief Use this template class to generate a 4 channel PixelID.
102 \description Use this template class to generate a 4 channel PixelID (64-bit identifier for
103  a pixel format used throughout PVR Assets from the channel information.
104  Simply define the template parameters for your class and get the ID member.
105  EXAMPLE USE: \code uint64 myPixelID = GeneratePixelType4<'b','g','r','a',8,8,8,8>::ID; \endcode
106 \param C1Name The Name of the 1st channel (poss. values 'r','g','b','a','l',0)
107 \param C2Name The Name of the 2nd channel (poss. values 'r','g','b','a','l',0)
108 \param C3Name The Name of the 3rd channel (poss. values 'r','g','b','a','l',0)
109 \param C4Name The Name of the 4th channel (poss. values 'r','g','b','a','l',0)
110 \param C1Bits The number of bits of the 1st channel
111 \param C2Bits The number of bits of the 2nd channel
112 \param C3Bits The number of bits of the 3rd channel
113 \param C4Bits The number of bits of the 4th channel
114 ********************************************************************************************/
115 template <char8 C1Name, char8 C2Name, char8 C3Name, char8 C4Name,
116  uint8 C1Bits, uint8 C2Bits, uint8 C3Bits, uint8 C4Bits>
117 class GeneratePixelType4
118 {
119 public:
120  static const uint64 ID =
121  (static_cast<uint64>(C1Name) + (static_cast<uint64>(C2Name) << 8) +
122  (static_cast<uint64>(C3Name) << 16) + (static_cast<uint64>(C4Name) << 24) +
123  (static_cast<uint64>(C1Bits) << 32) + (static_cast<uint64>(C2Bits) << 40) +
124  (static_cast<uint64>(C3Bits) << 48) + (static_cast<uint64>(C4Bits) << 56));
125 };

202 \param z Offset Z
203 \param width Texture Width

206 inline uint64 textureOffset3D(uint64 x, uint64 y, uint64 z, uint64 width, uint64 height)
