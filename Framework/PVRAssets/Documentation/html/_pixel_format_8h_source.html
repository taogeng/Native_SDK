55  /*!***********************************************************************************************************
56  \brief Returns the "content", or "name" of a channel, as a character. (normally r,g,b,a,d,s,l,i)
57  \param channel The zero-indexed channel of the texture(0, 1, 2, 3)
58  \return Return a character describing the channel contents
59  \description For example, the format d24s8 would return 'd' for channel:0, 's' for channel:1, NULL otherwise
60  *************************************************************************************************************/
61  char getChannelContent(uint8 channel)
62  {
63  if (channel >= 4) { return 0; }
64  return m_format.m_pixelTypeChar[channel];
65  }
66 
67  /*!***********************************************************************************************************
68  \brief Get the width of the specified channel
69  \param channel The zero-indexed channel of the texture(0, 1, 2, 3)
70  \return Return The number of bits the specified channel takes up.
71  *************************************************************************************************************/
72  uint8 getChannelBits(uint8 channel)
73  {
74  if (channel > 4) { return 0; }
75  return m_format.m_pixelTypeChar[channel + 4];
76  }
77 
78  /*!***********************************************************************************************************
79  \brief Get the number of channels in the format.
80  \return Return the number of channels in the format.
81  *************************************************************************************************************/
82  uint8 getNumberOfChannels()
83  {
84  return
85  m_format.m_pixelTypeChar[7] ? 4 :
86  m_format.m_pixelTypeChar[6] ? 3 :
87  m_format.m_pixelTypeChar[5] ? 2 :
88  m_format.m_pixelTypeChar[4] ? 1 : 0;
89  }
90 
91  /*!***********************************************************************************************************
92  \brief Returns true if the format is a "normal" compressed format, i.e. the format is not regular (channel type/
93  bitrate combination), but excludes some special packed formats that are not compressed, such as shared exponent
94  formats.
95  *************************************************************************************************************/
96  uint8 isCompressedFormat()
97  {
98  return (m_format.Part.High == 0) && (m_format.Part.Low != CompressedPixelFormat::SharedExponentR9G9B9E5);
99  }
100 
101  /*!***********************************************************************************************************
102  \brief Returns if the format is some kind of directly supported format that is not regular (i.e. channel type/
103  channel bitrate combination). I.e. returns true if the format is any of the formats described in the supported
104  "compressed" formats enumeration.
105  *************************************************************************************************************/
106  uint8 isIrregularFormat()
107  {
108  return m_format.Part.High == 0;
109  }
