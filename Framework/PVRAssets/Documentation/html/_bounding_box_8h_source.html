17 /*!*********************************************************************************************************************
18 \brief Return bounding box from vertex data.
19 \param[in] data ?ertex data
20 \param[in] stride_bytes Vertex stride in bytes
21 \param[in] offset_bytes Offset to the vertex data
22 \param[in] size_bytes Data size
23 \return The Axis-aligned bounding box of the data
24 ***********************************************************************************************************************/
25 inline math::AxisAlignedBox getBoundingBox(const byte* data, size_t stride_bytes, size_t offset_bytes, size_t size_bytes)
26 {
27  math::AxisAlignedBox aabb;
28  assertion(data);
29  assertion(stride_bytes >= 12 || !stride_bytes);
30  assertion(size_bytes >= stride_bytes);
31  if (size_bytes && data)
32  {
33  data = data + offset_bytes;
34  glm::vec3 minvec; glm::vec3 maxvec;
35  memcpy(&minvec, data, 12);
36  memcpy(&maxvec, data, 12);
37 
38  for (size_t i = stride_bytes; i < size_bytes; i += stride_bytes)
39  {
40  glm::vec3 vec; memcpy(&vec, data + i, 12);
41  minvec = glm::min(vec, minvec);
42  maxvec = glm::max(vec, maxvec);
43  }
44  aabb.setMinMax(minvec, maxvec);
45  }
46  else
47  {
48  aabb.setMinMax(glm::vec3(0, 0, 0), glm::vec3(0, 0, 0));
49  }
50 }
