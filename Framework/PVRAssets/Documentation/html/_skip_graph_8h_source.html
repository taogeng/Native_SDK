185  \brief Searches through the hash table to see if the added node already

188  bool addNode(const T& data)
189  {
190  size_t hashed = hash<string>()(data->toString());
191  ptrdiff_t iArrayElement(-1);
192 
193  // First, search the hash table to see
194  // if the node already exists.
195  SkipGraphNode<T>* skipGraphNode(findNode(hashed));
196  if (skipGraphNode == NULL)
197  {
198  // The node wasn't found, so a new node needs to be
199  // created.
200  m_aHashTable.push_back(HashElement(hashed, data));
201  iArrayElement = m_aHashTable.size() - 1;
202 
203  // Now point to the new instance.
204  skipGraphNode = &m_aHashTable[iArrayElement].getNode();
205  }
206  return skipGraphNode ? true : false;
207  }

300  /*!***************************************************************************
301  \brief Use the input hash to search the hash table and see if the
302  node already exists. If it does, the function returns a pointer
303  to the node. If it doesn't, it returns NULL.
304  \param[in] hash The hash value to search with
305  \return Return a handle to the found node, else return NULL
306  *****************************************************************************/
307  SkipGraphNode<T>* findNode(const size_t hash)
308  {
309  int i(0);
310  int i32HashTableSize((int32)m_aHashTable.size());
311 
312  // A NULL hash means the node has not initialized
313  // correctly.
314  if (hash == 0) { return NULL; }
315 
316  // NOTE:
317  // In the future, the hash table could be sorted from
318  // lowest hash value to highest so that binary search could
319  // be used to find a given node. It should be possible to
320  // use a bool (or some other mechanism) to toggle this form of search
321  // (if the skip graph is small, it might be faster to just use a brute
322  // force for loop to search through).
323  for (i = 0; i < i32HashTableSize; ++i)
324  {
325  if (m_aHashTable[i].getHash()) { return &m_aHashTable[i].getNode(); }
326  }
327 
328  // The element wasn't found, so return null.
329  return NULL;
330  }
