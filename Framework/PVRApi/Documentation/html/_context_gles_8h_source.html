256  /*!*********************************************************************************************************************
257  \brief Internal use. State tracking. enable veretx atribute.
258  ***********************************************************************************************************************/
259  void enableAttribute(uint16 attributeIdx)
260  {
261  m_renderStatesTracker.attributesToEnableBitfield |= (1 << attributeIdx);
262  if ((m_renderStatesTracker.attributesEnabledBitfield & (1 << attributeIdx)) == 0)
263  {
264  gl::EnableVertexAttribArray(attributeIdx);
265  m_renderStatesTracker.attributesEnabledBitfield |= (1 << attributeIdx);
266  }
267  m_renderStatesTracker.attributesMaxToEnable = (std::max)(attributeIdx, m_renderStatesTracker.attributesMaxToEnable) + 1;
268  }
269 
270  /*!*********************************************************************************************************************
271  \brief Internal use. State tracking.disable unneeded vertex attributes.
272  ***********************************************************************************************************************/
273  void disableUnneededAttributes()
274  {
275  uint32 deactivate = m_renderStatesTracker.attributesEnabledBitfield ^ m_renderStatesTracker.attributesToEnableBitfield;
276 
277  for (uint8 i = 0; i < m_renderStatesTracker.attributesMaxEnabled; ++i)
278  {
279  if (Bitfield<uint32>::isSet(deactivate, i))
280  {
281  gl::DisableVertexAttribArray(i);
282  }
283  }
284  m_renderStatesTracker.attributesEnabledBitfield = m_renderStatesTracker.attributesToEnableBitfield;
285  m_renderStatesTracker.attributesMaxEnabled = m_renderStatesTracker.attributesMaxToEnable;
286  m_renderStatesTracker.attributesMaxToEnable = 0;
287  m_renderStatesTracker.attributesToEnableBitfield = 0;
288  }
