348 /*!****************************************************************************************************************
349 \brief Get the display's color, depth and or stencil format from a DisplayAttributes object.
350 \param[in] attribute The DisplayAttributes to query. Normally from shell.getDisplayAttributes()
351 \param[out] outColorFmt The Color Format that this DisplayAttributes object has
352 \param[out] outDepthStencilFmt The DepthStencilFormat that this DisplayAttributes object has
353 \description This function is usually called on an object found with Shell::getDisplayAttributes(), in order to
354 determine the formats that are necessary for an On Screen FBO.
355 *******************************************************************************************************************/
356 inline void getDisplayFormat(const system::DisplayAttributes& attribute,
357  api::ImageDataFormat* outColorFmt,
358  api::ImageDataFormat* outDepthStencilFmt)
359 {
360  if (outColorFmt)
361  {
362  outColorFmt->format = PixelFormat(attribute.redBits ? 'r' : 0, attribute.greenBits ? 'g' : 0,
363  attribute.blueBits ? 'b' : 0, attribute.alphaBits ? 'a' : 0,
364  (uint8)attribute.redBits, (uint8)attribute.greenBits,
365  (uint8)attribute.blueBits, (uint8)attribute.alphaBits);
366  outColorFmt->colorSpace = attribute.frameBufferSrgb ? types::ColorSpace::sRGB : types::ColorSpace::lRGB;
367  }
368 
369  if (outDepthStencilFmt)
370  {
371  outDepthStencilFmt->format = PixelFormat('d', (attribute.stencilBPP ? 's' : 0), (uint8)0, (uint8)0,
372  (uint8)attribute.depthBPP, (uint8)attribute.stencilBPP, (uint8)0, (uint8)0);
373  outDepthStencilFmt->colorSpace = types::ColorSpace::lRGB;
374  outDepthStencilFmt->dataType = (uint8)attribute.depthBPP == 16 ? VariableType::UnsignedShortNorm : (uint8)attribute.depthBPP == 24 ? VariableType::UnsignedInteger : VariableType::Float;
375  }
376 }
